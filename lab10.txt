echo Check File

fileName='friends'
if [ -f $fileName ]
then
  if [ ! -w $fileName ]
  then
    chmod 700 $fileName
  fi
  echo Found $fileName
  more $fileName
  date >> $fileName
  d=$(date)
  newName="$fileName$d"
  mv $fileName "$newName"
  chmod 400 "$newName"

  echo File not found.
